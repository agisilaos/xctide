xctide - wrapper around xcodebuild with TUI and machine-friendly modes

USAGE:
  xctide [flags] [-- <xcodebuild args>]
  xctide build [flags] [-- <xcodebuild args>]
  xctide run [flags] [-- <xcodebuild args>]
  xctide plan [flags] [-- <xcodebuild args>]
  xctide doctor [--json]
  xctide destinations [flags] [--json]
  xctide xcrun <args...>
  xctide xctest <args...>

FLAGS:
  -h, --help            Show this help
      --version         Print version and exit
      --scheme string
      --workspace string
      --project string
      --configuration string
      --destination string
      --platform string  Destination filter for `destinations`
      --simulator-only   Filter to simulator destinations (`destinations`)
      --device-only      Filter to physical device destinations (`destinations`)
      --progress string  Progress mode: auto|tui|plain|json|ndjson
      --result-bundle string
      --plain           Disable TUI and stream raw output
      --json            Emit JSON summary to stdout
      --quiet           Pass -quiet to xcodebuild
      --verbose         Print wrapper diagnostics to stderr
      --no-input        Never prompt for selection
      --no-color        Disable color output

ENV:
  XCTIDE_SCHEME, XCTIDE_WORKSPACE, XCTIDE_PROJECT, XCTIDE_CONFIGURATION, XCTIDE_DESTINATION, XCTIDE_PROGRESS

EXAMPLES:
  xctide
  xctide build --scheme Subsmind --destination "platform=iOS Simulator,name=iPhone 16"
  xctide run --scheme Subsmind --destination "platform=iOS Simulator,id=<UDID>"
  xctide plan --scheme Subsmind -- test
  xctide doctor
  xctide destinations --scheme Subsmind
  xctide xcrun simctl list devices available
  xctide xcrun xctrace list templates
  xctide xctest -h
  xctide --plain -- test
  xctide --progress plain -- test
  xctide --progress json -- test
  xctide --progress ndjson -- test
